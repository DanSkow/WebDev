# Go to https://github.com/F5Networks/f5-telemetry-streaming/releases and get the full download URL and version number

bash
cd /var/config/rest/downloads

# Dynamic Variables
Version=v1.27.1-1
DownloadURL=https://github.com/F5Networks/f5-telemetry-streaming/releases/download/v1.27.1/f5-telemetry-1.27.1-1.noarch.rpm

# Static Variables
FN=f5-telemetry-$Version.noarch.rpm
CREDS=admin

#Download RPM File
curl -LJ -o /var/config/rest/downloads/$FN $DownloadURL

#Install RPM File
curl -kvu $CREDS -X POST http://localhost:8100/mgmt/shared/iapp/package-management-tasks -d "{\"operation\":\"INSTALL\",\"packageFilePath\":\"/var/config/rest/downloads/$FN\"}"
